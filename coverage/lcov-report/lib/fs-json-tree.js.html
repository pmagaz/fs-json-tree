'use strict';
 
var fs = require('fs');
 
var fsJsonTree = module.exports = {};
 
fsJsonTree.create = function(data, callback){
  this.parseData(data);
}
 
fsJsonTree.parseData = function(data, prev){
  for (var prop in data){
    var prev = data.name;
    if (data[prop] instanceof Array){
      for (var i = 0; i < data[prop].length;i++){
        var item = data[prop][i];
        this.itemHandler(item, prev);
        this.parseData(item, prev);
      }
    }
  }
}
 
fsJsonTree.itemHandler = function(item, prev){
  if (this.isDirectory(item.content)){
    this.directoryHandler(item, prev);
  } else{
    this.fileHandler(item, prev);
  }
}
 
fsJsonTree.isDirectory = function(directory){
  return directory instanceof Array;
}
 
fsJsonTree.fileHandler = function(file, prev){
